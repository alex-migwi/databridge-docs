End-to-End Testing

Prerequisites

Before testing, you need:

  • MySQL database running
  • Tables created in database
  • DataBridge CLI built and ready

Setting Up MySQL with Docker

Quick start with Docker Compose:

# docker/docker-compose.yml
version: '3.8'

services:
  mysql:
    image: mysql:8.0
    container_name: databridge-mysql
    environment:
      MYSQL_ROOT_PASSWORD: password
      MYSQL_DATABASE: testdb
    ports:
      - "3306:3306"
    volumes:
      - mysql-data:/var/lib/mysql

volumes:
  mysql-data:

Start MySQL:

cd docker
docker-compose up -d

# Verify it's running
docker ps

import { FastifyInstance } from 'fastify';

export default async function registerRoutes(fastify: FastifyInstance) {
  // List all products
  fastify.get('/products', async (request, reply) => {
    const items = await fastify.prisma.products.findMany();
    return items;
  });

  // Get product by ID
  fastify.get('/products/:id', async (request, reply) => {
    const { id } = request.params as { id: string };
    const item = await fastify.prisma.products.findUnique({
      where: { id: parseInt(id) },
    });
    
    if (!item) {
      reply.code(404);
      return { error: 'Not found' };
    }
    
    return item;
  });

  // Create product
  fastify.post('/products', async (request, reply) => {
    const item = await fastify.prisma.products.create({
      data: request.body,
    });
    reply.code(201);
    return item;
  });

  // Update product
  fastify.patch('/products/:id', async (request, reply) => {
    const { id } = request.params as { id: string };
    const item = await fastify.prisma.products.update({
      where: { id: parseInt(id) },
      data: request.body,
    });
    return item;
  });

  // Delete product
  fastify.delete('/products/:id', async (request, reply) => {
    const { id } = request.params as { id: string };
    await fastify.prisma.products.delete({
      where: { id: parseInt(id) },
    });
    reply.code(204);
  });
}

Common Issues and Solutions

Issue 1: Connection Error

Error:

‚ùå Failed to introspect database:
Can't reach database server at `localhost:3306`

Solutions:

  1. Verify MySQL is running: docker ps
  2. Check port is correct: 3306
  3. Test connection: mysql -h localhost -u root -ppassword testdb
  4. Check firewall settings

Issue 2: Authentication Failed

Error:

Error: P1000: Authentication failed against database server

Solutions:

  1. Verify credentials in .databridge.json
  2. Check .env file has correct DATABASE_URL
  3. Test login: mysql -u root -ppassword

Issue 3: No Models Found

Error:

No models found in schema. Run "databridge introspect" first.

Solutions:

  1. Run databridge introspect to generate schema
  2. Verify prisma/schema.prisma has models
  3. Check database has tables: SHOW TABLES;

Issue 4: Port Already in Use

Error:

Error: listen EADDRINUSE: address already in use :::3000

Solutions:

  1. Kill process on port 3000: lsof -ti:3000 | xargs kill -9 (Mac/Linux)
  2. Change port in .databridge.json: "port": 3001
  3. Find and stop other server: netstat -ano | findstr :3000 (Windows)

Issue 5: TypeScript Compilation Errors

Error:

Property 'prisma' does not exist on type 'FastifyInstance'

Solutions:

  1. Generate Prisma client: npx prisma generate
  2. Restart TypeScript server in VS Code
  3. Add type declaration file (see below)

Create types/fastify.d.ts:

import { PrismaClient } from '@prisma/client';

declare module 'fastify' {
  interface FastifyInstance {
    prisma: PrismaClient;
  }
}

Creating an Example Angular App

Setup Angular Application

# Install Angular CLI
npm install -g @angular/cli

# Create new Angular app
ng new example-angular-app
cd example-angular-app

# Copy generated services
cp -r ../src/app/databridge ./src/app/

Using Generated Services

app.component.ts:

import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { HttpClientModule } from '@angular/common/http';
import { ProductsService } from './databridge/services/products.service';
import { Product } from './databridge/models/product.model';

@Component({
  selector: 'app-root',
  standalone: true,
  imports: [CommonModule, HttpClientModule],
  template: `
    <div class="container">
      <h1>Products</h1>
      
      <div *ngIf="loading">Loading...</div>
      
      <div *ngIf="error" class="error">{{ error }}</div>
      
      <div class="products" *ngIf="!loading">
        <div class="product-card" *ngFor="let product of products">
          <h2>{{ product.name }}</h2>
          <p>{{ product.description }}</p>
          <p class="price">\${{ product.price }}</p>
          <p class="stock">Stock: {{ product.stock }}</p>
          <button (click)="deleteProduct(product.id)">Delete</button>
        </div>
      </div>
      
      <div class="add-product">
        <h2>Add Product</h2>
        <input [(ngModel)]="newProduct.name" placeholder="Name" />
        <input [(ngModel)]="newProduct.description" placeholder="Description" />
        <input [(ngModel)]="newProduct.price" type="number" placeholder="Price" />
        <input [(ngModel)]="newProduct.stock" type="number" placeholder="Stock" />
        <button (click)="addProduct()">Add</button>
      </div>
    </div>
  `,
  styles: [`
    .container { max-width: 1200px; margin: 0 auto; padding: 20px; }
    .products { display: grid; grid-template-columns: repeat(3, 1fr); gap: 20px; }
    .product-card { border: 1px solid #ddd; padding: 15px; border-radius: 8px; }
    .price { font-size: 24px; font-weight: bold; color: #2ecc71; }
    .error { color: red; padding: 10px; background: #fee; border-radius: 4px; }
  `]
})
export class AppComponent implements OnInit {
  products: Product[] = [];
  loading = true;
  error = '';
  
  newProduct: Partial<Product> = {
    name: '',
    description: '',
    price: 0,
    stock: 0,
  };

  constructor(private productsService: ProductsService) {}

  ngOnInit() {
    this.loadProducts();
  }

  loadProducts() {
    this.loading = true;
    this.productsService.list().subscribe({
      next: (data) => {
        this.products = data;
        this.loading = false;
      },
      error: (err) => {
        this.error = 'Failed to load products: ' + err.message;
        this.loading = false;
      }
    });
  }

  addProduct() {
    this.productsService.create(this.newProduct).subscribe({
      next: (product) => {
        this.products.push(product);
        this.newProduct = { name: '', description: '', price: 0, stock: 0 };
      },
      error: (err) => {
        this.error = 'Failed to add product: ' + err.message;
      }
    });
  }

  deleteProduct(id: number) {
    if (!confirm('Delete this product?')) return;
    
    this.productsService.delete(id).subscribe({
      next: () => {
        this.products = this.products.filter(p => p.id !== id);
      },
      error: (err) => {
        this.error = 'Failed to delete product: ' + err.message;
      }
    });
  }
}
