Build DataBridge from scratch - Part 2 covers the generate command, code generation with ts-morph, and server implementation

DataBridge CLI Tutorial - Part 2: Code Generation & Server

Continued from Part 1: Setup, Init, and Introspect

Command 3: Generate

Purpose

This is the heart of DataBridge - it generates:

  • API Routes: Fastify endpoints with CRUD operations
  • Angular Services: TypeScript services with HttpClient
  • Type Definitions: Shared TypeScript interfaces

Key Concepts

1. Prisma DMMF (Data Model Meta Format)

DMMF is like a JSON blueprint of your database schema.

import { getDMMF } from '@prisma/internals';

// Read schema.prisma file
const schemaContent = await fs.readFile('prisma/schema.prisma', 'utf-8');

// Convert to JSON structure
const dmmf = await getDMMF({ datamodel: schemaContent });

// dmmf structure:
{
  datamodel: {
    models: [
      {
        name: "Product",
        fields: [
          { name: "id", type: "Int", isRequired: true },
          { name: "name", type: "String", isRequired: true },
          { name: "price", type: "Float", isRequired: false }
        ]
      }
    ]
  }
}

Why use DMMF instead of regex?

  • Accurate: Understands Prisma syntax completely
  • Structured: Clean JSON format
  • Rich data: Includes relations, defaults, indexes

2. ts-morph (TypeScript AST Manipulation)

AST = Abstract Syntax Tree (the code structure as a tree)

Think of it as programmatically writing code:

import { Project, VariableDeclarationKind, Scope } from 'ts-morph';

// Create a TypeScript project in memory
const project = new Project();

// Add a new file
const sourceFile = project.createSourceFile('routes/products.ts', '', {
  overwrite: true,
});

// Add import statement
sourceFile.addImportDeclaration({
  namedImports: ['FastifyInstance'],
  moduleSpecifier: 'fastify',
});

// Add function
sourceFile.addFunction({
  name: 'registerRoutes',
  isAsync: true,
  parameters: [
    { name: 'fastify', type: 'FastifyInstance' },
  ],
  statements: `
    fastify.get('/products', async () => {
      return { message: 'List products' };
    });
  `,
});

// Save to disk
await sourceFile.save();

Why ts-morph?

  • Generates valid TypeScript automatically
  • Handles formatting, imports, syntax
  • No string concatenation nightmares!

3. Template Patterns

We use templates for repetitive code structures:

private generateRouteStatements(modelName: string): string {
  const lowerModel = modelName.toLowerCase();
  
  return `
    // GET all ${modelName}
    fastify.get('/${lowerModel}', async (request, reply) => {
      const items = await fastify.prisma.${lowerModel}.findMany();
      return items;
    });

    // GET one ${modelName} by ID
    fastify.get('/${lowerModel}/:id', async (request, reply) => {
      const { id } = request.params as { id: string };
      const item = await fastify.prisma.${lowerModel}.findUnique({
        where: { id: parseInt(id) },
      });
      
      if (!item) {
        reply.code(404);
        return { error: 'Not found' };
      }
      
      return item;
    });

    // POST create ${modelName}
    fastify.post('/${lowerModel}', async (request, reply) => {
      const item = await fastify.prisma.${lowerModel}.create({
        data: request.body,
      });
      reply.code(201);
      return item;
    });

    // PATCH update ${modelName}
    fastify.patch('/${lowerModel}/:id', async (request, reply) => {
      const { id } = request.params as { id: string };
      const item = await fastify.prisma.${lowerModel}.update({
        where: { id: parseInt(id) },
        data: request.body,
      });
      return item;
    });

    // DELETE ${modelName}
    fastify.delete('/${lowerModel}/:id', async (request, reply) => {
      const { id } = request.params as { id: string };
      await fastify.prisma.${lowerModel}.delete({
        where: { id: parseInt(id) },
      });
      reply.code(204);
    });
  `.trim();
}

The Generation Flow

1. Read DMMF from schema.prisma
   ‚Üì
2. For each model (User, Product, etc.):
   a. Generate API route file (routes/user.ts)
   b. Generate Angular service (services/user.service.ts)
   c. Generate TypeScript interface (models/user.model.ts)
   ‚Üì
3. Save all files to disk

Complete Implementation

// packages/cli/src/commands/generate.ts
import { Command } from '@oclif/core';
import { getDMMF } from '@prisma/internals';
import * as fs from 'fs-extra';
import * as path from 'path';
import chalk from 'chalk';
import { Project, VariableDeclarationKind, Scope } from 'ts-morph';

export default class Generate extends Command {
  static description = 'Generate API routes and Angular SDK from database schema';

  async run(): Promise<void> {
    this.log(chalk.bold.blue('\n‚öôÔ∏è  Generating code from schema...\n'));

    const config = await this.loadConfig();
    const dmmf = await this.getPrismaDMMF();

    if (!dmmf.datamodel.models.length) {
      this.error('No models found in schema. Run "databridge introspect" first.');
    }

    await this.generateApiRoutes(dmmf, config.api.outputPath);
    await this.generateAngularSdk(dmmf, config.angular.outputPath);

    this.log(chalk.green('\n‚úÖ Code generation complete!'));
    this.log(chalk.bold(`üì¶ Generated code for ${dmmf.datamodel.models.length} models`));
  }

  private async loadConfig() {
    const configPath = path.join(process.cwd(), '.databridge.json');
    if (!(await fs.pathExists(configPath))) {
      this.error('No .databridge.json found. Run "databridge init" first.');
    }
    return fs.readJSON(configPath);
  }

  private async getPrismaDMMF() {
    const schemaPath = path.join(process.cwd(), 'prisma', 'schema.prisma');
    
    if (!(await fs.pathExists(schemaPath))) {
      this.error('No schema.prisma found. Run "databridge introspect" first.');
    }

    const schemaContent = await fs.readFile(schemaPath, 'utf-8');
    return getDMMF({ datamodel: schemaContent });
  }

  private async generateApiRoutes(dmmf: any, outputPath: string) {
    this.log(chalk.cyan('Generating API routes...'));

    const routesDir = path.join(process.cwd(), outputPath, 'routes');
    await fs.ensureDir(routesDir);

    const project = new Project();

    for (const model of dmmf.datamodel.models) {
      const fileName = `${model.name.toLowerCase()}.ts`;
      const filePath = path.join(routesDir, fileName);

      const sourceFile = project.createSourceFile(filePath, '', {
        overwrite: true,
      });

      // Add imports
      sourceFile.addImportDeclaration({
        namedImports: ['FastifyInstance'],
        moduleSpecifier: 'fastify',
      });

      // Add route registration function
      sourceFile.addFunction({
        name: 'registerRoutes',
        isAsync: true,
        isExported: true,
        isDefaultExport: true,
        parameters: [
          { name: 'fastify', type: 'FastifyInstance' },
        ],
        statements: this.generateRouteStatements(model.name),
      });

      await sourceFile.save();
      this.log(chalk.green(`  ‚úì ${fileName}`));
    }
  }

  private generateRouteStatements(modelName: string): string {
    const lowerModel = modelName.toLowerCase();
    
    return `
      // List all ${modelName}
      fastify.get('/${lowerModel}', async (request, reply) => {
        const items = await fastify.prisma.${lowerModel}.findMany();
        return items;
      });

      // Get ${modelName} by ID
      fastify.get('/${lowerModel}/:id', async (request, reply) => {
        const { id } = request.params as { id: string };
        const item = await fastify.prisma.${lowerModel}.findUnique({
          where: { id: parseInt(id) },
        });
        
        if (!item) {
          reply.code(404);
          return { error: 'Not found' };
        }
        
        return item;
      });

      // Create ${modelName}
      fastify.post('/${lowerModel}', async (request, reply) => {
        const item = await fastify.prisma.${lowerModel}.create({
          data: request.body,
        });
        reply.code(201);
        return item;
      });

      // Update ${modelName}
      fastify.patch('/${lowerModel}/:id', async (request, reply) => {
        const { id } = request.params as { id: string };
        const item = await fastify.prisma.${lowerModel}.update({
          where: { id: parseInt(id) },
          data: request.body,
        });
        return item;
      });

      // Delete ${modelName}
      fastify.delete('/${lowerModel}/:id', async (request, reply) => {
        const { id } = request.params as { id: string };
        await fastify.prisma.${lowerModel}.delete({
          where: { id: parseInt(id) },
        });
        reply.code(204);
      });
    `.trim();
  }

  private async generateAngularSdk(dmmf: any, outputPath: string) {
    this.log(chalk.cyan('\nGenerating Angular SDK...'));

    const servicesDir = path.join(process.cwd(), outputPath, 'services');
    const modelsDir = path.join(process.cwd(), outputPath, 'models');
    
    await fs.ensureDir(servicesDir);
    await fs.ensureDir(modelsDir);

    const project = new Project();

    // Generate models (TypeScript interfaces)
    await this.generateModels(dmmf, modelsDir, project);

    // Generate services
    for (const model of dmmf.datamodel.models) {
      const fileName = `${model.name.toLowerCase()}.service.ts`;
      const filePath = path.join(servicesDir, fileName);

      const sourceFile = project.createSourceFile(filePath, '', {
        overwrite: true,
      });

      // Add Angular imports
      sourceFile.addImportDeclaration({
        namedImports: ['Injectable'],
        moduleSpecifier: '@angular/core',
      });

      sourceFile.addImportDeclaration({
        namedImports: ['HttpClient'],
        moduleSpecifier: '@angular/common/http',
      });

      sourceFile.addImportDeclaration({
        namedImports: ['Observable'],
        moduleSpecifier: 'rxjs',
      });

      sourceFile.addImportDeclaration({
        namedImports: [model.name],
        moduleSpecifier: `../models/${model.name.toLowerCase()}.model`,
      });

      // Add @Injectable decorator and service class
      const serviceClass = sourceFile.addClass({
        name: `${model.name}Service`,
        isExported: true,
        decorators: [
          {
            name: 'Injectable',
            arguments: ['{ providedIn: \'root\' }'],
          },
        ],
      });

      // Add private http property
      serviceClass.addProperty({
        name: 'baseUrl',
        scope: Scope.Private,
        isReadonly: true,
        initializer: `'http://localhost:3000/${model.name.toLowerCase()}'`,
      });

      // Add constructor
      serviceClass.addConstructor({
        parameters: [
          {
            name: 'http',
            type: 'HttpClient',
            scope: Scope.Private,
          },
        ],
      });

      // Add CRUD methods
      this.addServiceMethods(serviceClass, model.name);

      await sourceFile.save();
      this.log(chalk.green(`  ‚úì ${fileName}`));
    }
  }

  private async generateModels(dmmf: any, modelsDir: string, project: Project) {
    for (const model of dmmf.datamodel.models) {
      const fileName = `${model.name.toLowerCase()}.model.ts`;
      const filePath = path.join(modelsDir, fileName);

      const sourceFile = project.createSourceFile(filePath, '', {
        overwrite: true,
      });

      // Generate interface
      const interfaceDecl = sourceFile.addInterface({
        name: model.name,
        isExported: true,
      });

      for (const field of model.fields) {
        interfaceDecl.addProperty({
          name: field.name,
          type: this.mapPrismaTypeToTs(field.type),
          hasQuestionToken: !field.isRequired,
        });
      }

      await sourceFile.save();
    }
  }

  private mapPrismaTypeToTs(prismaType: string): string {
    const typeMap: Record<string, string> = {
      Int: 'number',
      Float: 'number',
      String: 'string',
      Boolean: 'boolean',
      DateTime: 'Date',
      Json: 'any',
    };

    return typeMap[prismaType] || 'any';
  }

  private addServiceMethods(serviceClass: any, modelName: string) {
    const lowerModel = modelName.toLowerCase();

    // list()
    serviceClass.addMethod({
      name: 'list',
      returnType: `Observable<${modelName}[]>`,
      statements: `return this.http.get<${modelName}[]>(this.baseUrl);`,
    });

    // getById()
    serviceClass.addMethod({
      name: 'getById',
      parameters: [{ name: 'id', type: 'number' }],
      returnType: `Observable<${modelName}>`,
      statements: `return this.http.get<${modelName}>(\`\${this.baseUrl}/\${id}\`);`,
    });

    // create()
    serviceClass.addMethod({
      name: 'create',
      parameters: [{ name: 'data', type: `Partial<${modelName}>` }],
      returnType: `Observable<${modelName}>`,
      statements: `return this.http.post<${modelName}>(this.baseUrl, data);`,
    });

    // update()
    serviceClass.addMethod({
      name: 'update',
      parameters: [
        { name: 'id', type: 'number' },
        { name: 'data', type: `Partial<${modelName}>` },
      ],
      returnType: `Observable<${modelName}>`,
      statements: `return this.http.patch<${modelName}>(\`\${this.baseUrl}/\${id}\`, data);`,
    });

    // delete()
    serviceClass.addMethod({
      name: 'delete',
      parameters: [{ name: 'id', type: 'number' }],
      returnType: 'Observable<void>',
      statements: `return this.http.delete<void>(\`\${this.baseUrl}/\${id}\`);`,
    });
  }
}

Command 4: Serve

Purpose

Start a Fastify web server that:

  • Auto-discovers and registers generated routes
  • Enables CORS for Angular app
  • Provides health check endpoint
  • Creates Prisma plugin for database access

Key Concepts

1. Fastify Framework

Fastify is one of the fastest Node.js web frameworks:

import Fastify from 'fastify';

const fastify = Fastify({
  logger: {
    transport: {
      target: 'pino-pretty',  // Pretty console logs
      options: { colorize: true },
    },
  },
});

// Start server
await fastify.listen({ port: 3000, host: '0.0.0.0' });

Why Fastify?

  • Fast: ~30% faster than Express
  • Modern: Built-in async/await support
  • Type-safe: Great TypeScript support
  • Plugin system: Modular architecture

2. CORS (Cross-Origin Resource Sharing)

Allows Angular app to call API from different origin:

import cors from '@fastify/cors';

await fastify.register(cors, {
  origin: '*',  // Allow all origins (config from .databridge.json)
});

Without CORS:

Angular (http://localhost:4200) ‚Üí API (http://localhost:3000)
‚ùå Browser blocks request: "No 'Access-Control-Allow-Origin' header"

With CORS:

Angular (http://localhost:4200) ‚Üí API (http://localhost:3000)
‚úÖ API adds header: Access-Control-Allow-Origin: *

3. Dynamic Route Registration (glob)

Automatically find and load route files:

import { glob } from 'glob';

// Find all .ts/.js files in routes directory
const routeFiles = await glob('**/*.{ts,js}', {
  cwd: routesDir,
  absolute: true,
});

// [ '/path/to/routes/product.ts', '/path/to/routes/user.ts' ]

// Import and register each route
for (const file of routeFiles) {
  const routeModule = await import(file);
  const registerFn = routeModule.default;
  
  await registerFn(fastify);
}

Why glob?

  • Scales automatically (100 models = 100 routes auto-loaded)
  • No manual import statements
  • Works with any file structure

4. Fastify Plugins

Plugins extend Fastify with new functionality:

import fp from 'fastify-plugin';
import { PrismaClient } from '@prisma/client';

const prismaPlugin = fp(async (fastify, options) => {
  const prisma = new PrismaClient();
  
  // Add prisma to fastify instance
  fastify.decorate('prisma', prisma);
  
  // Cleanup on shutdown
  fastify.addHook('onClose', async () => {
    await prisma.$disconnect();
  });
});

export default prismaPlugin;

Now routes can use fastify.prisma.product.findMany()

Complete Implementation

// packages/cli/src/commands/serve.ts
import { Command } from '@oclif/core';
import Fastify from 'fastify';
import cors from '@fastify/cors';
import { glob } from 'glob';
import * as fs from 'fs-extra';
import * as path from 'path';
import chalk from 'chalk';
import { Project } from 'ts-morph';

export default class Serve extends Command {
  static description = 'Start the API server with auto-registered routes';

  async run(): Promise<void> {
    this.log(chalk.bold.blue('\nüöÄ Starting DataBridge API Server...\n'));

    const config = await this.loadConfig();
    
    const fastify = Fastify({
      logger: {
        transport: {
          target: 'pino-pretty',
          options: {
            colorize: true,
            translateTime: 'HH:MM:ss',
            ignore: 'pid,hostname',
          },
        },
      },
    });

    await this.registerPlugins(fastify, config);
    await this.registerRoutes(fastify, config.api.outputPath);
    await this.registerHealthCheck(fastify);

    try {
      const port = config.api.port || 3000;
      await fastify.listen({ port, host: '0.0.0.0' });
      
      this.log(chalk.green(`\n‚úÖ Server running at http://localhost:${port}`));
      this.log(chalk.cyan(`üìä Health check: http://localhost:${port}/health\n`));
    } catch (error: any) {
      this.error(chalk.red(`\n‚ùå Failed to start server: ${error.message}\n`));
    }
  }

  private async loadConfig() {
    const configPath = path.join(process.cwd(), '.databridge.json');
    if (!(await fs.pathExists(configPath))) {
      this.error('No .databridge.json found. Run "databridge init" first.');
    }
    return fs.readJSON(configPath);
  }

  private async registerPlugins(fastify: any, config: any) {
    // Enable CORS
    await fastify.register(cors, {
      origin: config.api.cors || '*',
    });

    // Load Prisma plugin
    const pluginPath = path.join(
      process.cwd(),
      config.api.outputPath,
      'plugins',
      'prisma.ts'
    );

    if (!(await fs.pathExists(pluginPath))) {
      this.log(chalk.yellow('‚ö†Ô∏è  Prisma plugin not found. Creating...'));
      await this.createPrismaPlugin(config.api.outputPath);
    }

    const prismaPlugin = await import(pluginPath);
    await fastify.register(prismaPlugin.default);

    this.log(chalk.green('‚úì Plugins registered\n'));
  }

  private async registerRoutes(fastify: any, outputPath: string) {
    const routesDir = path.join(process.cwd(), outputPath, 'routes');

    if (!(await fs.pathExists(routesDir))) {
      this.warn('No routes directory found. Run "databridge generate" first.');
      return;
    }

    // Find all route files
    const routeFiles = await glob('**/*.{ts,js}', {
      cwd: routesDir,
      absolute: true,
    });

    if (!routeFiles.length) {
      this.warn('No route files found. Run "databridge generate" first.');
      return;
    }

    // Register each route
    let registeredCount = 0;
    for (const file of routeFiles) {
      try {
        const routeModule = await import(file);
        const registerFn = routeModule.default;

        if (typeof registerFn === 'function') {
          await registerFn(fastify);
          registeredCount++;
          
          const routeName = path.basename(file, path.extname(file));
          this.log(chalk.green(`  ‚úì /${routeName}`));
        }
      } catch (error: any) {
        this.warn(`Failed to load route ${file}: ${error.message}`);
      }
    }

    this.log(chalk.bold(`\nüì¶ Registered ${registeredCount} route(s)\n`));
  }

  private async registerHealthCheck(fastify: any) {
    fastify.get('/health', async () => {
      return {
        status: 'ok',
        timestamp: new Date().toISOString(),
        uptime: process.uptime(),
      };
    });
  }

  private async createPrismaPlugin(outputPath: string) {
    const pluginsDir = path.join(process.cwd(), outputPath, 'plugins');
    await fs.ensureDir(pluginsDir);

    const project = new Project();
    const filePath = path.join(pluginsDir, 'prisma.ts');

    const sourceFile = project.createSourceFile(filePath, '', {
      overwrite: true,
    });

    sourceFile.addImportDeclaration({
      defaultImport: 'fp',
      moduleSpecifier: 'fastify-plugin',
    });

    sourceFile.addImportDeclaration({
      namedImports: ['PrismaClient'],
      moduleSpecifier: '@prisma/client',
    });

    sourceFile.addStatements(`
const prismaPlugin = fp(async (fastify, options) => {
  const prisma = new PrismaClient();
  
  fastify.decorate('prisma', prisma);
  
  fastify.addHook('onClose', async () => {
    await prisma.$disconnect();
  });
});

export default prismaPlugin;
    `.trim());

    await sourceFile.save();
    this.log(chalk.green('  ‚úì Created prisma.ts plugin\n'));
  }
}
